00086 //! Skip the JSON white spaces in a stream.
00087 /*! \param stream A input stream for skipping white spaces.
00088         \note This function has SSE2/SSE4.2 specialization.
00089 */
00090 template<typename Stream>
00091 void SkipWhitespace(Stream& stream) {
00092         Stream s = stream;      // Use a local copy for optimization
00093         while (s.Peek() == ' ' || s.Peek() == '\n' || s.Peek() == '\r' || s.Peek() == '\t')
00094                 s.Take();
00095         stream = s;
00096 }

00203         /*! \tparam parseFlags Combination of ParseFlag. 
00204                  \tparam Stream Type of input stream.

00209         */

00211         bool Parse(Stream& stream, Handler& handler) {
00212                 parseError_ = 0;
00213                 errorOffset_ = 0;
00214 
00215                 if (setjmp(jmpbuf_)) {
00216                         stack_.Clear();
00217                         return false;
00218                 }
00219 
00220                 SkipWhitespace(stream);
00221 
00222                 if (stream.Peek() == '\0')
00223                         RAPIDJSON_PARSE_ERROR("Text only contains white space(s)", stream.Tell());
00224                 else {
00225                         switch (stream.Peek()) {
00226                                 case '{': ParseObject<parseFlags>(stream, handler); break;
00227                                 case '[': ParseArray<parseFlags>(stream, handler); break;
00228                                 default: RAPIDJSON_PARSE_ERROR("Expect either an object or array at root", stream.Tell()); return false;

00231 
00232                         if (stream.Peek() != '\0')
00233                                 RAPIDJSON_PARSE_ERROR("Nothing should follow the root object or array.", stream.Tell());
00234                 }

00350                 unsigned codepoint = 0;
00351                 for (int i = 0; i < 4; i++) {
00352                         Ch c = s.Take();
00353                         codepoint <<= 4;
00354                         codepoint += c;
00355                         if (c >= '0' && c <= '9')
00356                                 codepoint -= '0';

00358                                 codepoint -= 'A' - 10;

00360                                 codepoint -= 'a' - 10;

00362                                 RAPIDJSON_PARSE_ERROR("Incorrect hex digit after \\u escape", s.Tell() - 1);
00363                                 return 0;
00364                         }
